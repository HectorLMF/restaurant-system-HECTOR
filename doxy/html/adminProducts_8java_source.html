45 void loadDrinks() {
46 try {
47 List<Drink> drinks = apiClient.getAllDrinks();
48 for (Drink drink : drinks) {
49 modelDrink.addRow(new Object[]{
50 drink.getDrinksId(),
51 drink.getItemDrinks(),
52 drink.getDrinksPrice()
53 });
54 }
55 jTable1.setModel(modelDrink);
56 } catch (Exception ex) {
57 JOptionPane.showMessageDialog(null, "Error loading drinks: " + ex.getMessage());
58 }
59 }
60
61 void loadAppetizer() {
62 try {
63 List<Appetizer> appetizers = apiClient.getAllAppetizers();
64 for (Appetizer appetizer : appetizers) {
65 modelappetizers.addRow(new Object[]{
66 appetizer.getAppetizersId(),
67 appetizer.getItemAppetizers(),
68 appetizer.getAppetizersPrice()
69 });
70 }
71 jTable2.setModel(modelappetizers);
72 } catch (Exception ex) {
73 JOptionPane.showMessageDialog(null, "Error loading appetizers: " + ex.getMessage());
74 }
75 }
76
77 void loadmainCourse() {
78 try {
79 List<MainCourse> mainCourses = apiClient.getAllMainCourses();
80 for (MainCourse mainCourse : mainCourses) {
81 modelmaincourse.addRow(new Object[]{
82 mainCourse.getFoodId(),
83 mainCourse.getItemFood(),
84 mainCourse.getFoodPrice()
85 });
86 }
87 jTable3.setModel(modelmaincourse);
88 } catch (Exception ex) {
89 JOptionPane.showMessageDialog(null, "Error loading main courses: " + ex.getMessage());
