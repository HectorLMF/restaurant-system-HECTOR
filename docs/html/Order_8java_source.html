282 private double sumFromModel(DefaultTableModel model) {
283 double sum = 0.0;
284 for (int row = 0; row < model.getRowCount(); row++) {
285 Object qtyObj = model.getValueAt(row, 3); // Qty
286 Object priceObj = model.getValueAt(row, 2); // Price
287
288 if (qtyObj == null || priceObj == null)
289 continue;
290
291 int qty;
292 int price;
293 try {
294 qty = ((Number) qtyObj).intValue();
295 price = ((Number) priceObj).intValue();
296 } catch (ClassCastException e) {
297 // Fallback in case something comes as String
298 qty = Integer.parseInt(qtyObj.toString());
299 price = Integer.parseInt(priceObj.toString());
300 }
301
302 if (qty > 0) {
303 sum += qty * price;
304 }
305 }
306 return sum;
307 }

632 private void payBtnActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_payBtnActionPerformed
633 double itemsTotal = 0.0;
634
635 itemsTotal += sumFromModel(drinksModel);
636 itemsTotal += sumFromModel(appetizersModel);
637 itemsTotal += sumFromModel(mainsModel);
638
639 if (itemsTotal == 0.0) {
640 JOptionPane.showMessageDialog(
641 this,
642 "Please select at least one item (Qty > 0).",
643 "No items selected",
644 JOptionPane.INFORMATION_MESSAGE);
645 return;
646 }
647
648 // Calculate bill using the dedicated service
649 lastBill = orderService.calculateBill(itemsTotal);
650
651 subTotal = lastBill.getSubTotal();
652 vat = lastBill.getVat();
653 total = lastBill.getTotal();
654
655 subTotalLbl.setText("SubTotal: " + subTotal + " SR");
656 vatLbl.setText("VAT included: " + vat + " SR");
657 totalLbl.setText("Total: " + total + " SR");
658
659 JOptionPane.showMessageDialog(this, "Paid successfully");
660 }// GEN-LAST:event_payBtnActionPerformed
