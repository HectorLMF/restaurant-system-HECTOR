109 private <T> T execute(String action, IOCallable<T> callable) {
110 try {
111 return callable.call();
112 } catch (InterruptedException e) {
113 Thread.currentThread().interrupt();
114 throw new ApiClientException("Thread interrupted during " + action, e);
115 } catch (IOException e) {
116 throw new ApiClientException("I/O error during " + action, e);
117 }

124 @FunctionalInterface
125 private interface ResponseHandler<R> {
126 R handle(int status, String body) throws IOException;
127 }
128
134 private <R> R sendAndHandle(String action, HttpRequest request, ResponseHandler<R> handler)
135 throws InterruptedException, IOException {
136 HttpResponse<String> res = http.send(request, HttpResponse.BodyHandlers.ofString());
137 int status = res.statusCode();
138 String body = res.body();
139
140 LOGGER.info("{} -> HTTP {}", action, status);
141 LOGGER.info("Response body: '{}'", body);
142
143 try {
144 return handler.handle(status, body);
145 } catch (IOException e) {
146 // Wrap parsing IO exceptions with more context (status + body) so the UI
147 // receives a helpful message instead of a generic one.
148 throw new ApiClientException("I/O error during " + action + " status: " + status + " body: " + body, e);
149 }
150 }

169 private <T> T get(String path, Class<T> responseType) {
170 return execute("GET " + path, () -> sendAndHandle("GET " + path,
171 HttpRequest.newBuilder().uri(URI.create(baseUrl + path)).GET().header("Accept", APPLICATION_JSON).build(),
172 (status, body) -> {
173 if (status == 204 || body == null || body.trim().isEmpty()) {
174 return null;
175 }
176 if (status < 200 || status >= 300) {
177 throw new ApiClientException("GET " + path + " failed with HTTP " + status + " body: " + body);
178 }
179 return mapper.readValue(body, responseType);
180 }));
181 }

225 private <T, R> R post(String path, T body, Class<R> responseType) {
226 return execute("POST " + path, () -> {
227 String json = mapper.writeValueAsString(body);
228 HttpRequest req = HttpRequest.newBuilder().uri(URI.create(baseUrl + path))
229 .POST(HttpRequest.BodyPublishers.ofString(json)).header(CONTENT_TYPE, APPLICATION_JSON).build();
230
231 return sendAndHandle("POST " + path, req, (status, responseBody) -> {
232 if (status < 200 || status >= 300) {
233 throw new ApiClientException("POST " + path + " failed with HTTP " + status + " body: " + responseBody);
234 }
235 return mapper.readValue(responseBody, responseType);
236 });
237 });

572 String path = API_LOGIN;
573 return execute("login for user=" + username, () -> {
574 String jsonBody = mapper.writeValueAsString(Map.of(
575 "username", username,
576 "password", password));
577
578 HttpRequest request = HttpRequest.newBuilder().uri(URI.create(baseUrl + path))
579 .header(CONTENT_TYPE, APPLICATION_JSON).POST(HttpRequest.BodyPublishers.ofString(jsonBody)).build();
580
581 return sendAndHandle("POST " + path, request, (status, body) -> {
582 if (status == 200) {
583 return mapper.readValue(body, User.class);
584 } else {
585 throw new ApiClientException("Login failed with status: " + status + " body: " + body);
586 }
