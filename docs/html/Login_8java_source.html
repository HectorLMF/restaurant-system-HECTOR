214 private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_jButton1ActionPerformed
215 // Get username and password from input fields.
216 String usernameInput = usernametxt.getText();
217 String passwordInput = new String(jPasswordField1.getPassword());
218
219 jButton1.setEnabled(false);
220 jButton1.setText("Loading...");
221
222 new Thread(() -> {
223 try {
224 // Authenticate user through AuthService.
225 User loggedInUser = authService.authenticate(usernameInput, passwordInput);
226
227 SwingUtilities.invokeLater(() -> {
228
229 // Open the corresponding window based on user role. Use the
230 // injectable factories so tests can substitute frames.
231 if ("ADMIN".equalsIgnoreCase(loggedInUser.getRole())) {
232 adminSupplier.get().setVisible(true);
233 } else if ("CASHIER".equalsIgnoreCase(loggedInUser.getRole())) {
234 cashierFactory.apply(loggedInUser.getUsername()).setVisible(true);
235 } else {
236 messageDialog.showMessage(this, "Unknown Role: " + loggedInUser.getRole());
237 jButton1.setEnabled(true);
238 jButton1.setText(LOGIN_STRING);
239 return;
240 }
241
242 this.dispose();
243 });
244
245 } catch (Exception e) {
246 SwingUtilities.invokeLater(() -> {
247 messageDialog.showMessage(this, "Login failed: " + e.getMessage(), "Login Error", JOptionPane.ERROR_MESSAGE);
248
249 jButton1.setEnabled(true);
250 jButton1.setText(LOGIN_STRING);
251 });
252 }
253 }).start();
254 }// GEN-LAST:event_jButton1ActionPerformed
